# Development install — mirrors pyproject.toml extras so you don't have to
# maintain dependency lists in two places.
#
# Usage:
#   pip install -r requirements.txt   # core + vector + dev tools
#
# Or install specific extras directly:
#   pip install -e .              # core only
#   pip install -e ".[vector]"    # + ChromaDB / sentence-transformers
#   pip install -e ".[dev]"       # + pytest / coverage
#   pip install -e ".[all]"       # everything

-e ".[all,dev]"
